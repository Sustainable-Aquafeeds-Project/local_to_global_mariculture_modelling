1 Overview

This document matches salmonid farm locations to FAO fishing regions and country-level production data, checks that the spatial matching is consistent, and assigns model species to individual farms.

2 Match farm locations to FAO regions

Pull in farm locations and check that they look right.

Get farm locations for salmonids 
farm_locations <- file.path(input_farm_coords_path, "salmon_marine_farm_locations.qs") %>% qread()

farms_sf <- farm_locations %>% 
  st_as_sf(coords = c("X", "Y"), crs = "EPSG:4326")

p_bigmap_mercator +

Add FAO Major fishing regions to farm locations. Why? Because the production data is classified by country and FAO region. Basically, this ensures that some of the big two-coasted countries (namely the USA and Canada) have the appropriate number of farms on both coasts.

Get FAO fishing region information 
farms_sf_list <- farm_locations %>% 
  st_as_sf(coords = c("X", "Y"), crs = "EPSG:4326") %>% 
  mutate(row_num = row_number()) %>% 
  group_by(row_num) %>% 
  group_split()

FAO_shp <- file.path(bigdata_path, "fao/FAO_AREAS_CWP_NOCOASTLINE") %>% 
  read_sf() %>% 
  filter(F_LEVEL == "MAJOR") %>% 
  select(F_CODE)

# This is just the paired codes and countries - for later
FAO_code_key <- file.path(bigdata_path, "fao/FAO_AREAS_CWP_NOCOASTLINE") %>% 
  read_sf() %>% 
  filter(F_LEVEL == "MAJOR") %>% 
  distinct(F_CODE, NAME_EN)
Intersect FAO fishing area polygons with farm location points 
# For testing function
# this_farm <- farms_sf_list[[1]]

# define cores for parallel process
future::plan(
  strategy = "multisession", 
  workers = parallel::detectCores()-2
  )

# Intersect farm points with FAO fishing code polygons
farm_fao_intersect <- furrr::future_map(
  farms_sf_list, 
  function(this_farm){
    this_farm_fao_code <- st_intersection(this_farm, FAO_shp)
    return(this_farm_fao_code)
  },
  .progress = T
  ) %>% 
  bind_rows() %>% 
  select(-row_num) %>% 
  rename(fao_code = F_CODE)

qsave(farm_fao_intersect, here(input_farm_coords_path, "farm_locations_w_fishing_areas.qs"))

3 Get production data

Average production over years 2019:2023.

Load and average aquaculture production data over 2017-2023 
production <- file.path(species_path, "production", "aquaculture_quantity.qs") %>% 
  qread()
# range(production$period)

production <- production %>% 
  filter(between(period, 2017, 2023)) %>% 
  group_by(country, Scientific_Name, environment, ISO3_Code, fao_fishing_area) %>% 
  reframe(production = mean(value)) %>% 
  filter(production > set_units(0, "t"))

qsave(production, file.path(species_path, "production", "aquaculture_production_2017_2023.qs"))

production %>% 
  mutate(
    model_name = case_when(
      grepl("Salmo salar|Salmonidae|Salmo ", Scientific_Name) ~ "atlantic_salmon", 
      grepl("Oncorhynchus|Salmoniformes", Scientific_Name) ~ "general_salmonid",
      TRUE ~ NA

  filter(!is.na(model_name)) %>% 
  mutate(country = droplevels(country)) %>% 
  # Collapse fishing regions just for viewing
  group_by(country, model_name) %>% 
  reframe(production = sum(production))

4 Check matching

Check that mapped farm locations for each country are in the same FAO fishing regions as the production data 
# Get the mapped farm location data
farm_fao_intersect <- qread(file.path(input_farm_coords_path, "farm_locations_w_fishing_areas.qs"))

# Get production data
mariculture_prod_all <- file.path(gendata_path, "mariculture_production_all.qs") %>% 
  qread() %>% 
  filter(!str_detect(fao_fishing_area, "Inland"))

farm_list <- farm_fao_intersect %>% 
  group_by(iso3c, country, species_group) %>% 
  group_split()

# For testing function
# this_country_region <- farm_list[[4]]

fao_area_code_check <- map_df(
  farm_list, 
  function(this_country_region){

    # Get mapped values
    farm_country_name <- this_country_region$country %>% unique()
    farm_iso3c <- this_country_region$iso3c %>% unique()
    farm_fao_code <- this_country_region$fao_code %>% unique() %>% sort()
    farm_spp_grp <- this_country_region$species_group %>% unique()
    
    # Get production values
    prod_data <- mariculture_prod_all %>% 
      filter(
        ISO3_Code == farm_iso3c & 
        fao_code %in% farm_fao_code & 
        country == farm_country_name & 
        farm_name == farm_spp_grp
      )
    prod_spp_group <- prod_data$farm_name %>% unique()     
    prod_fao_code <- prod_data$fao_code %>% unique() %>% sort()
    
    # Return a tibble to check if they match
    tibble(
      iso3c = farm_iso3c, 
      countryname = farm_country_name, 
      farm_spp_group = farm_spp_grp, 
      mapped_fishing_areas = paste(farm_fao_code, collapse = ' '), 
      production_fishing_areas = paste(prod_fao_code, collapse = ' ')
      ) %>% 
      mutate(same_fishing_area = mapped_fishing_areas == production_fishing_areas)
  })

fao_area_code_check %>% 
  filter(!production_fishing_areas == "" & farm_spp_group == "Salmonidae fish") %>% 
  print(n = 50)

There’s a problem here: in the USA, there are salmonid farms mapped to fishing areas 21 (east coast) and 67 (west coast), but the production data only has it placed in 21 (east coast). How to solve this? We could do an extra step before moving on:

  • Assume that all USA farms produce the same amount each (both east and west coast)
  • Count the number of farms on the east and west coast in the US (from mapping data)
  • Modify the mariculture_production_all to split the production between the two fishing areas
Split USA production across east and west coast fishing regions proportionally 
tmp <- farm_fao_intersect %>% 
  filter(iso3c %in% c("USA", "CAN"))

# Just to make sure the farms are actually showing up
# p_bigmap_mercator +
#   geom_sf(
#     data = tmp, 
#     aes(color = country, geometry = geometry), 
#     size = 2
#   ) +
#   coord_sf(xlim = c(-70, -65), ylim = c(42, 46)) +
#   prettyplot_300() + theme(legend.position = "right")

EC_farms <- tmp[tmp$iso3c == "USA" & tmp$fao_code == 21, ] %>% nrow()
WC_farms <- tmp[tmp$iso3c == "USA" & tmp$fao_code == 67, ] %>% nrow()

# Add proportionate production for west coast
USA_prod_67 <- mariculture_prod_all %>% 
  filter(ISO3_Code == "USA") %>% 
  mutate(
    production = production * WC_farms/(EC_farms+WC_farms),
    fao_code = "67",
    fao_fishing_area = FAO_code_key$NAME_EN[FAO_code_key$F_CODE == "67"]
    )

# Adjust production for east coast
USA_prod_21 <- mariculture_prod_all %>% 
  filter(ISO3_Code == "USA") %>% 
  mutate(production = production * EC_farms/(EC_farms+WC_farms))

# Add back into overall production df
mariculture_prod_all <- rbind(
  mariculture_prod_all %>% filter(ISO3_Code != "USA"),
  USA_prod_67,
  USA_prod_21
)

# Overwrite previous file
qsave(mariculture_prod_all, file.path(gendata_path, "mariculture_production_all.qs"))

If you run the above check again the production should now be fixed.

5 Assign species to individual farms

Assign salmonid species to individual farm locations based on proportional production 
mariculture_prod_sal <- mariculture_prod_all %>% 
  filter(farm_name == "Salmonidae fish")

# E.g.
tmp <- mariculture_prod_sal %>% filter(country == "Australia")
nrow(tmp) # There is 1 species (model_name) included in salmonids in Australia
sum(tmp$production)/unique(tmp$nat_prod_all) # Salmonids make up 72.54% of all mariculture production
tmp$production[1]/unique(tmp$nat_prod_sal) # The first model_name listed (atlantic_salmon) makes up 100% of Australian salmonid production

tmp <- mariculture_prod_sal %>% filter(country == "Chile")
nrow(tmp) # There are 3 species (2 model_names) included in salmonids in Chile
sum(tmp$production)/unique(tmp$nat_prod_all) # Salmonids make up 70.54% of all mariculture production in Chile
tmp$production[3]/unique(tmp$nat_prod_sal) # The third model_name listed (atlantic_salmon) makes up 72% of Chilean salmonid production

# Finally, collapse by model_name (get rid of individual species)
mariculture_prod_sal <- mariculture_prod_sal %>% 
  group_by(ISO3_Code, country, fao_code, fao_fishing_area, farm_name, model_name, nat_prod_sal, nat_prod_all) %>% 
  reframe(production = sum(production))

qsave(mariculture_prod_sal, file.path(input_farm_coords_path, "mariculture_prod_salmonids.qs"))
