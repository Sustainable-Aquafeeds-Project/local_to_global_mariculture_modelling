3 Assign production to farms

What this does across each country, species group, and fishing region: - Counts the estimated number of farms (from mapping data) - Pulls the correct (matching) production data, groups by model_name (NOT farm_name), and estimates the number of farms that needs to be assigned to that model_name (based on proportionate salmonid production) - Assigns that model_name to a selection of farms in that country/region

# For testing
# this_country_region <- farm_list[[3]]

map_spp_to_locations_list <- map(
  farm_list, 
  function(this_country_region){
    # Determine which region we're looking at
    this_iso3c <- this_country_region$iso3c %>% unique()
    this_fao_code <- this_country_region$fao_code %>% unique()
    this_species_group <- this_country_region$species_group %>% unique()
    
    # Count how many farms need to be in that region (all salmonids)
    this_n_farms <- this_country_region %>% nrow()

    # Get the matching production data and calculate what proportion of national salmonid production should go to each model_name
    this_country_region_spp_production_list <- mariculture_prod_sal %>% 
      filter(
        ISO3_Code == this_iso3c, 
        farm_name == this_species_group, 
        fao_code == this_fao_code
        ) %>% 
      group_by(model_name, nat_prod_sal) %>% 
      reframe(production = sum(production)) %>% 
      mutate(prop_production = drop_units(production/nat_prod_sal)) %>% 
      distinct() %>% 
      group_by(model_name) %>% 
      group_split()

    # For testing:
    # this_species_prop <- this_country_region_spp_production_list[[1]]
    
    # For each model name is the country/fishing region, assign a random selection of farms to it (matching its proportionate production)
    farm_locations_w_model_spp <- map_df(
      this_country_region_spp_production_list, 
      function(this_species_prop){
        
        this_spp_n_farms <- round(this_n_farms * this_species_prop$prop_production)
        
        set.seed(1234)
        indices <- sample(nrow(this_country_region), size = this_spp_n_farms)
          
        this_spp_locations <- this_country_region[indices,]
        this_spp_locations$model_name <- this_species_prop$model_name

        this_country_region <- this_country_region[-indices,]
      
        return(this_spp_locations)
    })
    return(farm_locations_w_model_spp)
}, .progress = T
)

map_spp_to_locations <- lapply(
    map_spp_to_locations_list, 
    FUN = function(element) if(nrow(element) == 0) NULL else element
  ) %>% 
  bind_rows() %>% # removes the NULL record returns
  mutate(farm_id = row_number())

farm_static_data <- farms_w_temp_df %>%

farm_temporal_data <- farms_w_temp_df %>%

harvest_size <- purrr::map_dfr(
  farm_data, 
  function(temp) {
    # Grow "ideal" fish
    fg <- fish_growth(
      pop_params = all_params,
      species_params = all_params,
      water_temp = temp$temp_c,
      feed_params = feed_params[["plant_dominant_biomar"]],
      times = c(t_start = min(temp$day), t_end = max(temp$day), dt = 1),
      init_weight = all_params["meanW"],
      ingmax = all_params["meanImax"]
    ) %>%
      as.data.frame() %>% 
      mutate(farm_ID = unique(temp$farm_id))

    # Slice a "harvest window" and pick best time
    fg %>% 
      slice_tail(n = 60) %>% 
      filter(!is.na(weight)) %>% 
      slice_max(weight) %>% 
      select(farm_ID, days, weight)
  },
  .progress = T
)

6 Assign stocking population

Production needs to be assigned based on what the farm can actually handle. A country’s production is the sum of all its farms harvest size * stocking population. For the sake of simplicity, lets assume that each country/fao_code uses the same stocking population in each of its farms. Therefore, we need to find the correct stocking density (population) for the farms in each country so that it will meet its projected production.

\[ Prod_{national} = \sum{pop_{national} * indsize_{farm}} \]

combs <- farm_static_data %>% distinct(iso3c, fao_code)

farm_static_data <- file.path(input_farm_coords_path, "farm_static_data.qs") %>%
