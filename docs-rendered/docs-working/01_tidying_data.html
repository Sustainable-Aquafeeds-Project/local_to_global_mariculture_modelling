1 Aquaculture production

Clean aquaculture production data from the FAO.

# Get aquaculture quantity in tonnes live weight
quantity <- file.path(bigdata_path, "Aquaculture_2025.1.0", "Aquaculture_Quantity.csv") %>% 
  read.csv() %>% 
  clean_names() %>% 
  mutate(value = set_units(value, "t")) %>% 
  select(-measure)

# Apply environment codes and filter by "marine"
environment_codes <- file.path(bigdata_path, "Aquaculture_2025.1.0", "CL_FI_PRODENVIRONMENT.csv") %>% 
  read.csv() %>% 
  distinct(Code, Name_En)

quantity <- quantity %>% 
  merge(environment_codes, by.x = "environment_alpha_2_code", by.y = "Code") %>%
  rename(environment = Name_En) %>% 
  select(-environment_alpha_2_code)

# Apply species codes
species_codes <- file.path(bigdata_path, "Aquaculture_2025.1.0", "CL_FI_SPECIES_GROUPS.csv") %>% 
  read.csv() %>% 
  distinct(X3A_Code, Scientific_Name)

quantity <- quantity %>% 
  merge(species_codes, by.x = "species_alpha_3_code", by.y = "X3A_Code") %>% 
  filter(!is.na(Scientific_Name)) %>% 
  select(-species_alpha_3_code)

# Apply status codes
status_codes <- file.path(bigdata_path, "Aquaculture_2025.1.0", "CL_FI_SYMBOL_SDMX.csv") %>% 
  read.csv() %>% 
  distinct(Symbol, Name_En, Description_En)

quantity <- quantity %>% 
  merge(status_codes, by.x = "status", by.y = "Symbol") %>% 
  rename(status_nm = Name_En,
         status_description = Description_En) %>% 
  select(-status)

# Apply country and area codes
country_codes <- file.path(bigdata_path, "Aquaculture_2025.1.0", "CL_FI_COUNTRY_GROUPS.csv") %>% 
  read.csv() %>% 
  select(UN_Code, Identifier, Name_En, ISO3_Code)
area_codes <- file.path(bigdata_path, "Aquaculture_2025.1.0", "CL_FI_WATERAREA_GROUPS.csv") %>% 
  read.csv() %>% 
  select(Code, Name_En)

quantity <- quantity %>% 
  merge(country_codes, by.x = "country_un_code", by.y = "UN_Code", all.x = T, all.y = F) %>% 
  merge(area_codes, by.x = "area_code", by.y = "Code", all.x = T, all.y = F) %>% 
  rename(
    country = Name_En.x, 
    fao_fishing_area = Name_En.y,
    country_id = Identifier
    ) %>% 
  select(-status_description)

quantity <- quantity %>% 
  mutate(
    environment = as.factor(environment),
    Scientific_Name = as.factor(Scientific_Name),
    status_nm = as.factor(status_nm),
    country = as.factor(country),
    fao_fishing_area = as.factor(fao_fishing_area),
    ISO3_Code = as.factor(ISO3_Code)
  ) %>% 
  filter(
    period > 2000 &
    environment == "Marine" & 
    status_nm != "Not significant (<0.5)"
    )

qsave(quantity, file.path(species_path, "production", "aquaculture_quantity.qs"))

3 Daily SST data

From NASA - GHRSST Level 4 MUR 0.25deg Global Foundation Sea Surface Temperature Analysis

SST_files <- file.path(bigdata_path, "NASA/GHRSST_L4_MUR_0.25_SST") %>% 
  list.files(full.names = TRUE, pattern = ".nc")

# this_file <- SST_files[[1]]
# this_year=2013

# Initialise lists for each year
years <- c(2017:2022)
year_lists <- vector(mode ="list", length = length(years))

# Add annual files to a list of files for each year
for(y in 1:length(years)){
  this_year <- years[y]
  this_years_files <- SST_files[grep(paste0(this_year, "_"), SST_files)]
  this_years_files <- this_years_files[!grepl(paste0(this_year, "_0229"), this_years_files)]
  year_lists[[y]] <- this_years_files
}
  
# Extract the daily files for each year and rasterize in a stack
days <- c(1:365)
rast_list = vector(mode = "list", length = length(days))

map(
  days, 
  function(this_day){
    this_days_file_list <- sapply(year_lists, "[[", this_day)
    this_day_stack <- rast(this_days_file_list, subds = "analysed_sst")
    this_day_mean_rast <- app(this_day_stack, "mean", na.rm=TRUE) - 273.15
    writeRaster(
      this_day_mean_rast, 
      filename = file.path(
        input_farm_sst_path, "SST_raw_rasters", sprintf(
        "sst_nasa_mur_L4_0.25_mean_2017_2022_day_%s.tif", fixnum(this_day, digits = 3)
      ))
    )
  },
  .progress = T
)
raw_SST_files <- list.files(file.path(input_farm_sst_path, "SST_raw_rasters"), full.names = TRUE) %>% 
  sort()

# For testing function
# this_file <- raw_SST_files[[1]]

map(
  raw_SST_files, 
  function(this_file){
    saveName <-  basename(this_file)
    this_rast <- rast(this_file)
    this_gf_rast <- focal(this_rast, w = 7, fun = "mean", na.policy = "only")
    writeRaster(
      this_gf_rast, 
      file.path(input_farm_sst_path, "SST_gf_rasters", sprintf("%s", saveName)),
      overwrite = T
    )
  },
  .progress = T
)
