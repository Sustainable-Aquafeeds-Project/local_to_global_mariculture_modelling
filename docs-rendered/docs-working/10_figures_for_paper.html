2 Get data

farm_locs <- file.path(input_farm_coords_path, "farm_static_data.qs") %>% 
  qread() %>% 
  rename(farm_ID = farm_id) %>%
  mutate(
    longitude = st_coordinates(geometry)[, 1],
    latitude = st_coordinates(geometry)[, 2]
  ) %>% 
  mutate(
    country = as.factor(country),
    region = case_when(
      country %in% c("Canada", "United States of America") & longitude < -100 ~ "North America (WC)", 
      country %in% c("Canada", "United States of America") & longitude < -50 ~ "North America (EC)", 
      country %in% c("Ireland", "United Kingdom") ~ "UK & Ireland",
      country %in% c("Iceland", "Faroe Islands") ~ "Iceland & FI",
      country %in% c("Norway", "Finland") ~ "Scandinavia",
      country == "Australia" ~ "Australia",
      country == "Chile" ~ "Chile", 
      country == "Russian Federation" ~ NA
    ) %>% as.factor()
  )

The total values describe selected output variables (measures) summed over the entire period of a salmon growing period (548 days). The data were generated from a suite of model runs which cover several dimensions:

  • 2750 unique farms
  • 3 different feed groups, representing different formulations of salmon feed: MD (marine-dominant), AD (animal-dominant) and PD (plant-dominant)
  • 3 different feed types representing levels of digestibility of each of the three different feeds: mean digestibility, minimum digestibility, and maximum digestibility

3.2 C:N ratio to environment

farm_ends_CN_ratio <- rbind(
  farm_ends_carbon_uneat,
  farm_ends_carbon_excr,
  farm_ends_nitrogen_uneat,
  farm_ends_nitrogen_excr
) %>% 
  select(-c(mean_value, sd_value, sd_per_biom, n)) %>% 
  convert_feeds_to_levels() %>% 
  mutate(
    nutrient = case_when(str_detect(measure, "nitrogen") ~ "N", T ~ "C") %>% as.factor(),
    measure = str_remove_all(measure, "nitrogen_|carbon_"),
    mean_value_perbiom = mean_value_perbiom %>% set_units("g g_fish-1") %>% set_units("kg t_fish-1") %>% drop_units()
  )

farm_ends_CN_ratio <- rbind(
  farm_ends_CN_ratio,
  farm_ends_CN_ratio %>% 
    pivot_wider(names_from = measure, values_from = mean_value_perbiom) %>% 
    mutate(
      measure = "total",
      mean_value_perbiom = uneat + excr
    ) %>% 
    select(-c(uneat, excr))
) %>% 
  pivot_wider(names_from = nutrient, values_from = mean_value_perbiom) %>% 
  mutate(
    C = C/12.011,
    N = N/14.007,
    CN = C/N,
    measure = as.factor(measure)
  ) %>% 
  select(-c(C, N)) 
  
farm_ends_CN_ratio_means <- farm_ends_CN_ratio %>% 
  group_by(feed, feed_level, measure) %>% 
  reframe(
    CN_mean = mean(CN),
    CN_min = min(CN),
    CN_max = max(CN)
  )
farm_ends_CN_ratio_means %>%

  geom_point(size = 4, position = position_dodge(width = 0.75), colour = "black") +
  scale_fill_manual(values = feed_pal) +
  scale_shape_manual(values = c(
    "Excreted faeces" = 21, 
    "Total" = 24, 
    "Uneaten Feed" = 22
  )) +

4.2 Daily values

farm_runs <- map2_dfr(
  c(file.path(output_path, "model_run_outputs", "farm_runs_total_carbon.qs"),
    file.path(output_path, "model_run_outputs", "farm_runs_total_nitrogen.qs")),
  c("C", "N"),
  function(fnm, nutr) {
    fnm %>% 
      qread() %>% 
      filter(feed == "plant_dominant_biomar") %>% 
      select(-mean_value, sd_value) %>% 
      mutate(
        nutrient = as.factor(nutr),
        mean_value_perbiom = mean_value_perbiom %>% set_units("g g_fish-1") %>% set_units("kg t_fish-1") %>% drop_units(),
        sd_per_biom = sd_per_biom %>% set_units("g g_fish-1") %>% set_units("kg t_fish-1") %>% drop_units(),
      )
    }
  ) %>% 
  left_join(farm_locs, by = join_by(farm_ID))

# How many farms per country? 10 seems like enough
farm_runs <- farm_runs %>% 
  group_by(country, prod_t, nutrient) %>% 
  reframe(
    n = n(),
    mean = mean(mean_value_perbiom),
    sd = sqrt(mean(sd_per_biom^2))
  )
farm_runs %>% 
  filter(nutrient == "C") %>% 
  ggplot(aes(
    x = prod_t, 
    y = mean, 
    ymin = mean - sd, 
    ymax = mean + sd, 
    colour = country,
    fill = country
  )) +
  geom_line(linewdith = 0.75) +
  geom_ribbon(linewidth = 0.1, alpha = 0.25) +
  facet_wrap(~country, ncol = 2) +
  prettyplot_600() +
  labs(
    x = "Day of production",
    y = expression("Nutrient loading (kg"~t[fish]^{-1}*")")
)
