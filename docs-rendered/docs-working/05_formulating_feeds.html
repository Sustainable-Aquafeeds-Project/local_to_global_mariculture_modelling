1 Overview

The purpose of this document is to formulate all the feeds that will be used in the analysis. This needs to be done before the model pipeline is run.

2 Import ingredients

There is no check or correction here to ensure that protein + lipid + ash + carb = 1 in the incoming ingredient data. This needs to be checked manually prior to import.

Get ingredient details (PLC composition, digestibility) 
ingreds <- file.path(input_feed_profile_path, "all_ingredients.xlsx") %>% 
  read_excel(sheet = "all_ingredients") %>% 
  mutate(ingredient = as.factor(ingredient))

ingred_nms <- levels(ingreds$ingredient)

3 Import feeds

Unlike ingredients, the feeds do not need to be checked for sum = 1 prior to import. That check is done here.

Feed compositions 
feed_inputs <- file.path(input_feed_profile_path, "all_feeds.xlsx") %>% 
  read_excel(sheet = "all_feeds")
  
feed_inputs <- feed_inputs %>% 
  pivot_longer(names_to = "feed", values_to = "proportion", cols = !contains(c("ingredient", "category"))) %>% 
  mutate(
    feed = as.factor(feed),
    ingredient = as.factor(ingredient),
    proportion = case_when(is.na(proportion) ~ 0, T ~ proportion)
  ) %>% 
  left_join(ingreds, by = "ingredient")

Check that ingredient proportions sum to 1.

Check composition proportions sum to 1 
# Any missing ingredients?
feed_inputs %>% 
  filter(!str_detect(feed, regex("original", ignore_case = TRUE))) %>% 
  filter(str_detect(feed, regex("biomar", ignore_case = TRUE))) %>% 
  filter(proportion != 0 & is.na(protein)) %>% 
  pull(ingredient)

# Sum ingredients to 1
feed_inputs <- feed_inputs %>%
  filter(!str_detect(feed, regex("original", ignore_case = TRUE))) %>% 
  filter(!is.na(proportion) & proportion != 0 & str_detect(feed, regex("biomar", ignore_case = TRUE))) %>%
  group_by(feed) %>% 
  mutate(
    sum = sum(proportion),
    proportion = proportion/sum
  ) %>% 
  ungroup() %>% 
  mutate(
    ingredient = droplevels(ingredient),
    feed = droplevels(feed)
  ) %>% 
  select(-c(sum, ash, in_use, in_MS, exp_data, `SUM CHECK`))

4 Prepare for pipeline

The following chunk allows as many ingredients and feeds to be added to the incoming data as needed, it will still be formatted into a form that the targets pipelines can use.

Format feeds as lists to give to targets 
feed_types <- levels(feed_inputs$feed)

feed_params_all <- purrr::map(feed_types, function(ft) {
  # Filter by specified feed
  df <- feed_inputs %>% filter(feed == ft) 

  list(
    Proteins = df %>% 
      select(ingredient, proportion, contains("protein")) %>% 
      rename(
        macro = protein, 
        mean_digest = mean_protein_digestibility,
        min_digest = min_protein_digestibility,
        max_digest = max_protein_digestibility
      ),
    Carbohydrates = df %>% 
      select(ingredient, proportion, contains("carb")) %>% 
      rename(
        macro = carb, 
        mean_digest = mean_carb_digestibility,
        min_digest = min_carb_digestibility,
        max_digest = max_carb_digestibility
      ),
    Lipids = df %>% 
      select(ingredient, proportion, contains("lipid")) %>% 
      rename(
        macro = lipid, 
        mean_digest = mean_lipid_digestibility,
        min_digest = min_lipid_digestibility,
        max_digest = max_lipid_digestibility
      )
  )
}) %>% 
  setNames(feed_types)

This next chunk creates the mean, minimum and maximum digestibility variants of each of the feeds.

Split into mean, min, max digestibility 
# Functions
make_mean_digest <- function(df) {
  df %>% 
    mutate(digest = mean_digest) %>% 
    select(ingredient, proportion, macro, digest)
}
make_min_digest <- function(df) {
  df %>% 
    mutate(digest = min_digest) %>% 
    select(ingredient, proportion, macro, digest)
}
make_max_digest <- function(df) {
  df %>% 
    mutate(digest = max_digest) %>% 
    select(ingredient, proportion, macro, digest)
}

feed_params <- list(
  "marine_dominant_biomar"     = feed_params_all[["marine_dominant_biomar"]] %>% lapply(make_mean_digest),
  "marine_dominant_biomar_min" = feed_params_all[["marine_dominant_biomar"]] %>% lapply(make_min_digest),
  "marine_dominant_biomar_max" = feed_params_all[["marine_dominant_biomar"]] %>% lapply(make_max_digest),
  "animal_dominant_biomar"     = feed_params_all[["animal_dominant_biomar"]] %>% lapply(make_mean_digest),
  "animal_dominant_biomar_min" = feed_params_all[["animal_dominant_biomar"]] %>% lapply(make_min_digest),
  "animal_dominant_biomar_max" = feed_params_all[["animal_dominant_biomar"]] %>% lapply(make_max_digest),
  "plant_dominant_biomar"     = feed_params_all[["plant_dominant_biomar"]] %>% lapply(make_mean_digest),
  "plant_dominant_biomar_min" = feed_params_all[["plant_dominant_biomar"]] %>% lapply(make_min_digest),
  "plant_dominant_biomar_max" = feed_params_all[["plant_dominant_biomar"]] %>% lapply(make_max_digest)
  )

7 Comparing feeds

The three “biomar” feeds were designed to be isoenergetic.

Calculate energy stats about the feeds 
feed_params[[1]] %>% 
  lapply(function(ls) {
    ls %>% mutate(total = proportion * macro * digest) %>% 
      pull(total) %>% sum()
  }) %>% as.data.frame() %>% 
    mutate(
      Proteins = Proteins * 23.6,
      Carbohydrates = Carbohydrates * 17.2,
      Lipids = Lipids * 36.2,
      total = Proteins + Carbohydrates + Lipids
    )


feed_stats <- 
  map2(feed_params, names(feed_params), function(ls, feedname) {
    map2(ls, names(ls), function(df, macroname) {
      df %>% mutate(component = macroname)
    }) %>% bind_rows() %>% 
      mutate(feed = feedname)
  }) %>% 
  bind_rows() %>% 
  mutate(
    ing_macro = proportion * macro,
    digest_macro = ing_macro * digest,
    E = case_when(
      component == "Proteins" ~ digest_macro * 23.6,
      component == "Lipids" ~ digest_macro * 36.2,
      T ~ digest_macro * 17.2
    )
  ) %>% 
  group_by(feed) %>% 
  mutate(total_E = sum(E)) %>% 
  ungroup() %>% 
  group_by(feed, component, total_E) %>% 
  reframe(comp_E = sum(E)) %>% 
  mutate(prop_E = comp_E/total_E)

feed_stats %>% 
  group_by(feed) %>% 
  reframe(total_E = mean(total_E))

feed_stats %>% 
  filter(component == "Proteins")
